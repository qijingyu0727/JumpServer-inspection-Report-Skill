base_probe::sh -lc 'emit(){ sec="$1"; shift; printf "__JMS_BEGIN__ %s\n" "$sec"; eval "$*" 2>/dev/null || true; printf "\n__JMS_END__ %s\n" "$sec"; }; emit os_release "cat /etc/os-release"; emit sys_hostname "hostname"; emit sys_kernel "uname -r"; emit sys_arch "uname -m"; emit sys_now "date \"+%Y-%m-%d %H:%M:%S\""; emit sys_uptime "uptime"; emit app_version "jmsctl version || docker ps --format \"{{.Image}}\" | grep -m1 jumpserver | sed \"s#.*:##\" || podman ps --format \"{{.Image}}\" | grep -m1 jumpserver | sed \"s#.*:##\" || echo unknown"; emit cpuinfo "grep -E \"^(processor|physical id|cpu cores|model name)[[:space:]]*:\" /proc/cpuinfo"; emit mem_free "free -m"; emit disk "df -Th"'
security_probe::sh -lc 'emit(){ sec="$1"; shift; printf "__JMS_BEGIN__ %s\n" "$sec"; eval "$*" 2>/dev/null || true; printf "\n__JMS_END__ %s\n" "$sec"; }; os_id="$(awk -F= '\''/^ID=/{gsub(/"/,"",$2); print tolower($2); exit}'\'' /etc/os-release 2>/dev/null)"; os_like="$(awk -F= '\''/^ID_LIKE=/{gsub(/"/,"",$2); print tolower($2); exit}'\'' /etc/os-release 2>/dev/null)"; family="$os_id $os_like"; emit zombie_detail "ps -A -ostat=,ppid=,pid=,cmd= | awk '\''\$1 ~ /^[Zz]/ {print}'\''"; emit mem_top "ps -aux --sort=-%mem | head -n 11"; emit firewall "if printf \"%s\n\" \"$family\" | grep -Eq \"(ubuntu|debian)\"; then ufw status verbose | head -n 1 | awk '\''{print tolower(\$2)}'\''; elif printf \"%s\n\" \"$family\" | grep -Eq \"(centos|rhel|rocky|almalinux|anolis|ol|openeuler)\"; then firewall-cmd --state || echo inactive; else (nft list ruleset >/dev/null 2>&1 && echo active) || (iptables -L -n >/dev/null 2>&1 && echo active) || echo inactive; fi"; emit firewall_rules "if printf \"%s\n\" \"$family\" | grep -Eq \"(ubuntu|debian)\"; then ufw status verbose; elif printf \"%s\n\" \"$family\" | grep -Eq \"(centos|rhel|rocky|almalinux|anolis|ol|openeuler)\"; then firewall-cmd --list-all; else nft list ruleset | head -n 80 || iptables -L -n | head -n 80 || echo unknown; fi"'
container_probe::sh -lc 'emit(){ sec="$1"; shift; printf "__JMS_BEGIN__ %s\n" "$sec"; eval "$*" 2>/dev/null || true; printf "\n__JMS_END__ %s\n" "$sec"; }; emit jmsctl_status "jmsctl status | sed -n \"1,120p\" || echo unavailable"; emit containers "docker ps --format \"{{.Names}}\t{{.Status}}\t{{.Ports}}\" || podman ps --format \"{{.Names}}\t{{.Status}}\t{{.Ports}}\""; emit container_mem "docker stats --no-stream --format \"{{.Name}}\t{{.MemUsage}}\" || podman stats --no-stream --format \"{{.Name}}\t{{.MemUsage}}\""'
db_config_probe::sh -lc 'printf "__JMS_BEGIN__ db_config\n"; grep -E "^DB_(ENGINE|HOST|PORT|USER|PASSWORD|NAME)=" /opt/jumpserver/config/config.txt 2>/dev/null || true; printf "\n__JMS_END__ db_config\n"'
large_file_probe::sh -lc 'printf "__JMS_BEGIN__ large_files\n"; found=0; for d in /opt/jumpserver /data /data/docker; do [ -d "$d" ] && found=1; done; if [ "$found" -eq 1 ]; then if command -v timeout >/dev/null 2>&1; then timeout 8s sh -lc '\''find /opt/jumpserver /data /data/docker -xdev -type f -size +1G -print0 2>/dev/null | xargs -0 du -h 2>/dev/null | sort -h | tail -n 10'\''; else find /opt/jumpserver /data /data/docker -xdev -type f -size +1G -print0 2>/dev/null | xargs -0 du -h 2>/dev/null | sort -h | tail -n 10; fi; else echo -; fi; printf "\n__JMS_END__ large_files\n"'
